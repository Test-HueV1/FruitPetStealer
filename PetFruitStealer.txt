--// Drag Support
local dragging = false
local dragStart, startPos
local function updateDrag(input)
	local delta = input.Position - dragStart
	MainFrame.Position = UDim2.new(startPos.X.Scale, startPos.X.Offset + delta.X, startPos.Y.Scale, startPos.Y.Offset + delta.Y)
end
MainFrame.InputBegan:Connect(function(input)
	if input.UserInputType == Enum.UserInputType.MouseButton1 or input.UserInputType == Enum.UserInputType.Touch then
		dragging = true
		dragStart = input.Position
		startPos = MainFrame.Position
	end
end)
MainFrame.InputChanged:Connect(function(input)
	if dragging and (input.UserInputType == Enum.UserInputType.MouseMovement or input.UserInputType == Enum.UserInputType.Touch) then
		updateDrag(input)
	end
end)
userInput.InputEnded:Connect(function(input)
	if input.UserInputType == Enum.UserInputType.MouseButton1 or input.UserInputType == Enum.UserInputType.Touch then
		dragging = false
	end
end)

local function showNotification(message)
    NotificationLabel.Text = message
    NotificationLabel.Visible = true
    NotificationLabel.BackgroundTransparency = 0.5
    NotificationLabel.TextTransparency = 0

    local fadeInTweenInfo = TweenInfo.new(0.2, Enum.EasingStyle.Sine, Enum.EasingDirection.Out)
    local fadeOutTweenInfo = TweenInfo.new(0.5, Enum.EasingStyle.Sine, Enum.EasingDirection.Out, 0, false, 1.5)

    tweenService:Create(NotificationLabel, fadeInTweenInfo, {BackgroundTransparency = 0, TextTransparency = 0}):Play()

    task.wait(1)
    local fadeOutTween = tweenService:Create(NotificationLabel, fadeOutTweenInfo, {BackgroundTransparency = 1, TextTransparency = 1})
    fadeOutTween:Play()
    fadeOutTween.Completed:Wait()

    NotificationLabel.Visible = false
end

-- Function to populate the inventory frame
local function populateInventory(targetPlayer)
    -- Clear any existing buttons
    for _, child in pairs(InventoryFrame:GetChildren()) do
        if child:IsA("TextButton") then
            child:Destroy()
        end
    end

    if not targetPlayer then return end

    -- This is a placeholder. You need to access the player's actual inventory.
    -- For example, looking for a folder inside the player's character or a value in their leaderstats.
    local items = {} 
    
    -- Assuming items are a folder in the player's character or a value object
    if targetPlayer.Character then
        local petsFolder = targetPlayer.Character:FindFirstChild("Pets")
        if petsFolder then
            for _, pet in pairs(petsFolder:GetChildren()) do
                if pet:IsA("Model") then
                    table.insert(items, pet)
                end
            end
        end

        local fruitsFolder = targetPlayer.Character:FindFirstChild("Fruits")
        if fruitsFolder then
            for _, fruit in pairs(fruitsFolder:GetChildren()) do
                if fruit:IsA("BasePart") or fruit:IsA("Model") then
                    table.insert(items, fruit)
                end
            end
        end
    end

    for _, item in pairs(items) do
        local itemButton = Instance.new("TextButton")
        itemButton.Size = UDim2.new(0, 60, 0, 60)
        itemButton.Text = item.Name
        itemButton.Font = Enum.Font.GothamBold
        itemButton.TextSize = 10
        itemButton.TextColor3 = Color3.new(1, 1, 1)
        itemButton.BackgroundColor3 = Color3.fromRGB(60, 60, 70)
        Instance.new("UICorner", itemButton).CornerRadius = UDim.new(0, 8)
        itemButton.Parent = InventoryFrame

        itemButton.Activated:Connect(function()
            stealItem(targetPlayer, item)
        end)
    end
end
